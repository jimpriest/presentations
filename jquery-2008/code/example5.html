$(document).ready(function(){
	$("#verifyrequest").click(function() {
		var checked_status = this.checked;
		
		if(checked_status){
		    $("#verify").addClass("verify2");
		    $("#verify").fadeTo("slow", 0.7);
		    $(this).next().text("Congratulations!");
		} else {
		    $("#verify").removeClass("verify2").fadeTo("slow", 1);
		    $(this).next().text("You must volunteer!");
	     	}
		
		});
});
